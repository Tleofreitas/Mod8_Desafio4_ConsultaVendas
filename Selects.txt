-- SUMMARY
SELECT SEL.NAME, SUM(SAL.AMOUNT)
FROM TB_SALES SAL
JOIN TB_SELLER SEL ON SAL.SELLER_ID=SEL.ID
WHERE SAL.DATE BETWEEN '2022-01-01' AND '2022-06-30'
GROUP BY SEL.NAME

-- REPORT
SELECT SEL.ID, SAL.DATE, SAL.AMOUNT, SEL.NAME
FROM TB_SALES SAL
JOIN TB_SELLER SEL ON SAL.SELLER_ID=SEL.ID
WHERE SAL.DATE BETWEEN '2022-05-01' AND '2022-05-31' AND UPPER(SEL.NAME) LIKE UPPER('%odinson%')
